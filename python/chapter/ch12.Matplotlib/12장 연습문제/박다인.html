import csv
f = open('cafe_2year.csv', 'r', encoding='utf-8')
lines = csv.reader(f)

header = next(lines)

date_data = []
numCafe_data = []
monthCount = 1

for line in lines :
    if monthCount % 3 == 0 :
        date_data.append(line[0])
        numCafe_data.append(line[1])
    monthCount += 1

plt.bar(date_data, numCafe_data)

plt.title('전국의 월별 까페수',color='green')
plt.xlabel('년월')
plt.ylabel('카페수')

division_data = ['일반의','인턴','레지던트','전문의']

header = next(lines)
total = 0

    total += int(line[i+1])

    doctor_percent.append(percent)
