import csv

f = open('cafe_2year.csv', 'r', encoding='utf-8')
lines = csv.reader(f)             # 여기까지가 정보 불러오기

header = next(lines)        

xdata = []
ydata = []
count = 1

for line in lines:
    if count%3 == 0 :         # x 데이터를 3달마다 되었을 때
        xdata.append(line[0])  # xdata(=x축)에 lines에 있는 line을 0번째 인덱스 항목을 넣음
        ydata.append(line[1])  # ydata(=y축)에 lines에 있는 line을 1번째 인덱스 항목을 넣음

    count += 1        # 이걸 끝까지 돌아

font1 = {'size':20, 'color':'green'} 
    
plt.bar(xdata, ydata) # 그 데이터로 그래프를 그려

plt.title('전국의 월별 카페수', fontdict=font1)
plt.xlabel('년월')
plt.ylabel('카페수')

plt.show()

f = open('doctor2.csv', 'r', encoding='utf-8')
lines = csv.reader(f)

header = next(lines)

ydata2 = []
ydata3 = []
ydata4 = []

    xdata.append(line[0])
    ydata1.append(int(line[1]))
    ydata2.append(int(line[2]))
    ydata3.append(int(line[3]))
    ydata4.append(int(line[4]))

plt.legend(loc='best')

f = open('doctor2.csv', 'r', encoding='utf-8')
lines = csv.reader(f)

header = next(lines)


labels = ['일반의', '인턴', '레지던트', '전문의']
sizes = []
seoul = next(lines)

total = 0
for i in range(4) :
    total += int(seoul[i+1])
    
for i in range(4) :
    persent = int(seoul[i+1])/total;
    sizes.append(persent)


colors = ['blue', 'orange', 'green', 'red']
explode = (0, 0.1, 0, 0)
plt.title("서울 종합병원 의사 유형별 분포", color='green')
plt.pie(sizes, explode=(0, 0.1, 0, 0), labels=labels, colors=colors,
        autopct='%1.1f%%', shadow=True, startangle=90)
plt.axis('equal')
plt.show()
