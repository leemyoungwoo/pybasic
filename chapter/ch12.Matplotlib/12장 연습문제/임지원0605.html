    if is_first == True:
        is_first = False
        continue
    area.append(data[0])
    normal.append(int(data[1]))
    intern.append(int(data[2]))
    resident.append(int(data[3]))
    special.append(int(data[4]))

plt.plot(area,special, label='전문의', color ='purple', linestyle='-.', marker='d', linewidth=2, markersize=12)
