f = open('cafe_2year.csv','r',encoding='utf-8')
lines = csv.reader(f)
header = next(lines)

list_data = []
for line in lines :
    list_data.append(line[:])
f.close()

xdata = []
ydata = []
for i in range(0,len(list_data)):
    if i % 3 == 0:
        for j in range(0,2):
            if j == 0:
                xdata.append(list_data[i][j])
            elif j == 1:
                ydata.append(list_data[i][j])
            
plt.title('전국의 월별 카페 수', color='g', fontsize='30')
plt.bar(xdata,ydata)

plt.xlabel(header[0])
plt.ylabel(header[1])

plt.show()
