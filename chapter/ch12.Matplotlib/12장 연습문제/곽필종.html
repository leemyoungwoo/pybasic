import numpy as np
import csv

class Prac_Files:
    def __init__(self):
        self.f = open('doctor_2019.csv', 'r', encoding='utf-8')
        # self.location = location

    def header(self, f):
        lines = csv.reader(self.f)
        header = next(lines)
        data_set = np.array([header])
        data_set = np.delete(data_set, 1)
        return data_set

    def location_sum(self, f, location):
        data_set = [0, 0, 0, 0]
        lines = csv.reader(self.f)
        # header = next(lines)

        for line in lines:
            if line[0] == location:
                data_set[0] += int(line[2])
                data_set[1] += int(line[3])
                data_set[2] += int(line[4])
                data_set[3] += int(line[5])
        return data_set

    def close(self):
        self.f.close()
    


f = Prac_Files();
data =np.empty((0,5))
data = np.append(data, f.header(f))
location = ['서울', '부산', '대구', '인천', '대전', '광주', '울산']
for i in range(len(location)):
    f = Prac_Files();
    data = np.append(data, location[i])
    data = np.append(data, f.location_sum(f, location[i]))
data = data.reshape(8, 5).tolist()

for i in range(len(data)):
    for j in range(len(data[0])):
        print(data[i][j], end=' ')
    print()

x_data = []
y_data = [0,0,0,0]

with open('doctor2.csv', 'r', encoding='utf8') as f:
    lines = csv.reader(f)
    header = next(lines)
    for i in range(1, len(header)-1):
        x_data.append(header[i])
    
    for line in lines:
        if '서울' in line:
            y_data[0] += int(line[1])
            y_data[1] += int(line[2])
            y_data[2] += int(line[3])
            y_data[3] += int(line[4])
  
y_data2 = []
sum = y_data[0] + y_data[1] + y_data[2]+ y_data[3]
for i in range(len(y_data)):
    y_data2.append(y_data[i] /  sum * 100)

plt.title('서울 종합병원 의사 유형별 분포')
plt.pie(y_data2, explode = (0, 0.1, 0, 0), labels=x_data, autopct='%1.1f%%', shadow=True, startangle = 90)

plt.show()
