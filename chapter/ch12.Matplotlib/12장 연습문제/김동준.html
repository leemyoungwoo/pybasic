# 6. 4번 문제에서 사용한 doctor2.csv 파일을 읽어 서울 종합병원의 의사 수 분포를 의사 유형별로

import matplotlib.pyplot as plt
import numpy as np
import math as math
from matplotlib import rc

# Matplotlib 한글처리 코드
import matplotlib as mpl
import matplotlib.pyplot as plt
mpl.rcParams['axes.unicode_minus'] = False
mpl.rc('font', family='Malgun Gothic')
%matplotlib inline

f = open('doctor2.csv', 'r', encoding = 'utf-8')
lines = csv.reader(f)

header = next(lines)    #분리한 헤더는 legend로 넣기
#print(header)

x = []

sum = 0
lv = ['일반의','인턴','레지던트','전문의']
for line in lines:
    
    if line[0] == '서울' :
        x = line


header = x[0]
del (x[0])

intList1 = [int(_x) for _x in x]

sum = 0
for i in intList1:
    sum += i 

avgList1 = []
for i in range(4):
    x = intList1[i] / sum * 100
    
    avgList1.append(round(x,2))
#print(avgList1)

portion = []
for i in avgList1:
    portion.append(i)

plt.pie(portion, explode = (0 ,0.2, 0, 0), labels = lv, autopct = '%1.1f%%', shadow = False, startangle = 90)

plt.title('서울 종합병원 의사 유형별 분포')
